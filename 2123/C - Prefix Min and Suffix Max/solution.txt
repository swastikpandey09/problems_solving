for _ in range(int(ri())):
    n = int(ri())
    a = list(map(int, ri().split()))
    p = [a[0]]
    for i in range(1, n):
        p.append(min(p[-1], a[i]))
    s = [0] * n
    s[-1] = a[-1]
    for i in range(n - 2, -1, -1):
        s[i] = max(s[i + 1], a[i])
    ans = []
    for i in range(n):
        l = p[i - 1] if i else 10**9
        r = s[i + 1] if i < n - 1 else -10**9
        ans.append('1' if a[i] < l or a[i] > r else '0')
    print(''.join(ans))
